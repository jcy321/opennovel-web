```rust
/// 批注格式定义
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct AnnotationFormat {
    pub agent_name: String,
    pub content: String,
    pub timestamp: DateTime<Utc>,
    pub position: Option<TextPosition>,
}

impl AnnotationFormat {
    /// 转换为 HTML 注释格式
    pub fn to_html_comment(&self) -> String {
        format!(
            "<!-- 批注: {} | {} | {} -->",
            self.agent_name,
            self.content,
            self.timestamp.format("%Y-%m-%d %H:%M")
        )
    }
    
    /// 从 HTML 注释解析
    pub fn from_html_comment(comment: &str) -> Option<Self> {
        // 解析格式: <!-- 批注: Agent名 | 内容 | 时间 -->
        let re = regex::Regex::new(r"<!-- 批注: (.+?) \| (.+?) \| (.+?) -->").ok()?;
        let caps = re.captures(comment)?;
        
        Some(Self {
            agent_name: caps.get(1)?.as_str().to_string(),
            content: caps.get(2)?.as_str().to_string(),
            timestamp: DateTime::parse_from_rfc3339(caps.get(3)?.as_str())
                .ok()?
                .with_timezone(&Utc),
            position: None,
        })
    }
}
```

### 2.2 Agent 间 @提及机制

**目标**: 使 Agent 能够主动 @其他 Agent 触发协作

**新增模块**: `packages/sdk/core/src/collaboration/agent_mention.rs`

```rust
use serde::{Deserialize, Serialize};

/// Agent 提及请求
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct AgentMention {
    /// 发起者 Agent
    pub from_agent: String,
    /// 目标 Agent
    pub to_agent: String,
    /// 提及类型
    pub mention_type: MentionType,
    /// 关联的消息 ID
    pub message_id: String,
    /// 附带数据
    pub payload: Option<serde_json::Value>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum MentionType {
    /// 请求处理
    RequestAction,
    /// 传递批注
    PassAnnotation,
    /// 通知完成
    NotifyCompletion,
    /// 请求审阅
    RequestReview,
}

/// Agent 提及处理器
pub struct AgentMentionHandler {
    agent_registry: Arc<AgentRegistry>,
    message_queue: Arc<MessageQueue>,
}

impl AgentMentionHandler {
    /// 处理 Agent 提及
    pub async fn handle(&self, mention: AgentMention) -> Result<(), MentionError> {
        // 1. 检查目标 Agent 是否可用
        let agent = self.agent_registry.get_by_role(
            AgentRole::from_str(&mention.to_agent)?
        );
        
        if agent.is_none() {
            return Err(MentionError::AgentNotFound(mention.to_agent));
        }
        
        // 2. 创建群聊消息，标记 @提及
        let message = ChatMessage::agent_mention(
            &mention.from_agent,
            &mention.to_agent,
            mention.message_id.clone(),
        );
        
        // 3. 推送到消息队列
        self.message_queue.push(message).await;
        
        // 4. 触发目标 Agent 处理
        self.trigger_agent(&mention).await?;
        
        Ok(())
    }
    
    async fn trigger_agent(&self, mention: &AgentMention) -> Result<(), MentionError> {
        // 根据 mention_type 决定如何触发
        match mention.mention_type {
            MentionType::PassAnnotation => {
                // 传递批注给下一个 Agent
                // 例如：天道 -> 刘和平
            }
            MentionType::RequestReview => {
                // 请求审阅
            }
            _ => {}
        }
        Ok(())
    }
}
```

### 2.3 Agent 协作编排器

**目标**: 实现串行协作流程控制

**新增模块**: `packages/sdk/core/src/collaboration/workflow/orchestrator.rs`

```rust
use std::sync::Arc;
use async_trait::async_trait;

/// 协作编排器
pub struct CollaborationOrchestrator {
    agent_registry: Arc<AgentRegistry>,
    mention_handler: Arc<AgentMentionHandler>,
    state_store: Arc<WorkflowStateStore>,
}

/// 章节撰写工作流
pub struct ChapterWritingWorkflow {
    book_id: String,
    chapter_id: String,
    current_stage: WorkflowStage,
    history: Vec<WorkflowStep>,
}

#[derive(Debug, Clone)]
pub enum WorkflowStage {
    /// 天道推演
    TianDaoDerivation,
    /// 刘和平审阅人物
    LiuHepingReview,
    /// 世界观守护者审阅
    WorldGuardianReview,
    /// 天道调整
    TianDaoAdjustment,
    /// 用户确认
    UserConfirmation,
    /// 执笔撰写
    WriterWriting,
    /// 知识库更新
    KnowledgeUpdate,
}

impl CollaborationOrchestrator {
    /// 启动章节撰写工作流
    pub async fn start_chapter_workflow(
        &self,
        book_id: &str,
        chapter_id: &str,
    ) -> Result<ChapterWritingWorkflow, WorkflowError> {
        let workflow = ChapterWritingWorkflow {
            book_id: book_id.to_string(),
            chapter_id: chapter_id.to_string(),
            current_stage: WorkflowStage::TianDaoDerivation,
            history: Vec::new(),
        };
        
        // 步骤1: 天道推演
        self.execute_stage(&workflow, WorkflowStage::TianDaoDerivation).await?;
        
        Ok(workflow)
    }
    
    /// 执行工作流阶段
    async fn execute_stage(
        &self,
        workflow: &ChapterWritingWorkflow,
        stage: WorkflowStage,
    ) -> Result<(), WorkflowError> {
        match stage {
            WorkflowStage::TianDaoDerivation => {
                // 触发天道执行推演
                self.mention_handler.handle(AgentMention {
                    from_agent: "system".to_string(),
                    to_agent: "tiandao".to_string(),
                    mention_type: MentionType::RequestAction,
                    message_id: workflow.chapter_id.clone(),
                    payload: Some(serde_json::json!({
                        "action": "derive_plot",
                        "chapter_id": workflow.chapter_id,
                    })),
                }).await?;
            }
            
            WorkflowStage::LiuHepingReview => {
                // 天道 @刘和平 请求审阅
                self.mention_handler.handle(AgentMention {
                    from_agent: "tiandao".to_string(),
                    to_agent: "liuheping".to_string(),
                    mention_type: MentionType::PassAnnotation,
                    message_id: workflow.chapter_id.clone(),
                    payload: None,
                }).await?;
            }
            
            WorkflowStage::WorldGuardianReview => {
                // 刘和平 @世界观守护者 请求审阅
                self.mention_handler.handle(AgentMention {
                    from_agent: "liuheping".to_string(),
                    to_agent: "world_guardian".to_string(),
                    mention_type: MentionType::PassAnnotation,
                    message_id: workflow.chapter_id.clone(),
                    payload: None,
                }).await?;
            }
            
            WorkflowStage::TianDaoAdjustment => {
                // 世界观守护者 -> 天道调整
                self.mention_handler.handle(AgentMention {
                    from_agent: "world_guardian".to_string(),
                    to_agent: "tiandao".to_string(),
                    mention_type: MentionType::PassAnnotation,
                    message_id: workflow.chapter_id.clone(),
                    payload: None,
                }).await?;
            }
            
            WorkflowStage::UserConfirmation => {
                // 天道 @用户 请求确认
                // 发送群聊消息，等待用户响应
            }
            
            WorkflowStage::WriterWriting => {
                // 用户确认后，天道 @执笔
                self.mention_handler.handle(AgentMention {
                    from_agent: "tiandao".to_string(),
                    to_agent: "writer".to_string(),
                    mention_type: MentionType::RequestAction,
                    message_id: workflow.chapter_id.clone(),
                    payload: None,
                }).await?;
            }
            
            WorkflowStage::KnowledgeUpdate => {
                // 执笔 @各Agent 更新知识库
                for agent in &["observer", "world_guardian", "liuheping", "tiandao"] {
                    self.mention_handler.handle(AgentMention {
                        from_agent: "writer".to_string(),
                        to_agent: agent.to_string(),
                        mention_type: MentionType::NotifyCompletion,
                        message_id: workflow.chapter_id.clone(),
                        payload: None,
                    }).await?;
                }
            }
        }
        
        Ok(())
    }
    
    /// 推进到下一阶段
    pub async fn advance(
        &self,
        workflow: &mut ChapterWritingWorkflow,
    ) -> Result<(), WorkflowError> {
        let next_stage = self.get_next_stage(&workflow.current_stage)?;
        workflow.current_stage = next_stage.clone();
        self.execute_stage(workflow, next_stage).await
    }
    
    fn get_next_stage(&self, current: &WorkflowStage) -> Result<WorkflowStage, WorkflowError> {
        match current {
            WorkflowStage::TianDaoDerivation => Ok(WorkflowStage::LiuHepingReview),
            WorkflowStage::LiuHepingReview => Ok(WorkflowStage::WorldGuardianReview),
            WorkflowStage::WorldGuardianReview => Ok(WorkflowStage::TianDaoAdjustment),
            WorkflowStage::TianDaoAdjustment => Ok(WorkflowStage::UserConfirmation),
            WorkflowStage::UserConfirmation => Ok(WorkflowStage::WriterWriting),
            WorkflowStage::WriterWriting => Ok(WorkflowStage::KnowledgeUpdate),
            WorkflowStage::KnowledgeUpdate => Err(WorkflowError::WorkflowComplete),
        }
    }
}
```

```rust
use std::collections::HashMap;

/// 知识库维护者映射
pub struct KnowledgeMaintainers {
    mapping: HashMap<KnowledgeBaseType, AgentRole>,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum KnowledgeBaseType {
    Character,      // 人物数据库
    WorldView,      // 世界观数据库
    Timeline,       // 时间线
    Foreshadowing,  // 伏笔池
    Plot,           // 剧情知识库
    Map,            // 地图数据库
    Faction,        // 阵营派系
    ChapterContext, // 本章上下文
}

impl KnowledgeMaintainers {
    pub fn new() -> Self {
        let mut mapping = HashMap::new();
        
        // 根据设计文档分配
        mapping.insert(KnowledgeBaseType::Character, AgentRole::LiuHeping);
        mapping.insert(KnowledgeBaseType::WorldView, AgentRole::WorldGuardian);
        mapping.insert(KnowledgeBaseType::Timeline, AgentRole::Observer);
        mapping.insert(KnowledgeBaseType::Foreshadowing, AgentRole::TianDao);
        mapping.insert(KnowledgeBaseType::Plot, AgentRole::TianDao);
        mapping.insert(KnowledgeBaseType::Map, AgentRole::WorldGuardian);
        mapping.insert(KnowledgeBaseType::Faction, AgentRole::TianDao);
        mapping.insert(KnowledgeBaseType::ChapterContext, AgentRole::TianDao);
        
        Self { mapping }
    }
    
    /// 获取知识库的维护者
    pub fn get_maintainer(&self, kb_type: KnowledgeBaseType) -> Option<AgentRole> {
        self.mapping.get(&kb_type).copied()
    }
    
    /// 检查 Agent 是否有权限维护该知识库
    pub fn can_maintain(&self, agent: AgentRole, kb_type: KnowledgeBaseType) -> bool {
        self.mapping.get(&kb_type) == Some(&agent)
    }
}
```

```rust
    pub fn new() -> Self {

        ]));

            Permission::ReadKnowledge,
            Permission::WriteKnowledge,  // 限定于人物
            Permission::CreateAnnotation,
        ]));
        
        // 世界观守护者：可读写世界观、地图
        matrix.insert(AgentRole::WorldGuardian, HashSet::from([
            Permission::ReadKnowledge,
            Permission::WriteKnowledge,  // 限定于世界观、地图
            Permission::CreateAnnotation,
            Permission::Intervene,
        ]));
        
        // ... 其他 Agent
        
        Self { matrix }
    }
    
    /// 检查 Agent 对特定知识库的权限
    pub fn check_knowledge_permission(
        &self,
        role: AgentRole,
        kb_type: KnowledgeBaseType,
        permission: Permission,
    ) -> bool {
        let maintainers = KnowledgeMaintainers::new();
        
        // 写权限需要是维护者
        if permission == Permission::WriteKnowledge {
            return maintainers.can_maintain(role, kb_type);
        }
        
        // 读权限所有活跃 Agent 都有
        if permission == Permission::ReadKnowledge {
            return true;
        }
        
        false
    }
}
```

```rust
use axum::{Json, extract::Path};

/// 切换书籍阶段
pub async fn transition_stage(
    Path(book_id): Path<String>,
    Json(payload): Json<StageTransitionRequest>,
) -> Result<Json<StageTransitionResponse>, ApiError> {
    let transition_type = payload.transition_type;
    
    match transition_type {
        StageTransitionType::CompleteConception => {
            // 用户点击"构思完成"
            // 1. 锁定规划者
            // 2. 解锁其他 Agent
            // 3. 切换到撰写阶段
            trigger_conception_complete(&book_id).await?;
        }
        
        StageTransitionType::ForceTransition => {
            // 强制切换（管理员功能）
        }
    }
    
    Ok(Json(StageTransitionResponse {
        success: true,
        new_stage: BookStage::Writing,
    }))
}

async fn trigger_conception_complete(book_id: &str) -> Result<(), ApiError> {
    // 1. 获取群聊实例
    let group_chat = get_group_chat(book_id).await?;
    
    // 2. 切换阶段
    group_chat.change_stage(BookStage::Writing);
    
    // 3. 更新 Agent 状态
    // 规划者锁定
    group_chat.lock_agent("planner");
    // 调研者锁定
    group_chat.lock_agent("researcher");
    // 其他解锁
    for agent in &["tian_dao", "world_guardian", "liuheping", "writer", "reviewer", "observer"] {
        group_chat.unlock_agent(agent);
    }
    
    // 4. 发送系统消息
    let system_msg = ChatMessage::system(
        "阶段切换：进入撰写阶段\n规划者已锁定，天道已解锁"
    );
    group_chat.add_message(system_msg);
    
    // 5. 通知前端
    broadcast_stage_change(book_id, BookStage::Writing).await;
    
    Ok(())
}
```

```rust
impl ToolExecutor for CreateChapterTool {

        let title = input["title"].as_str().unwrap();

            "chapter_num": chapter_num,
            "title": title,

/// 写入章节梗概
pub struct WriteChapterOutlineTool;

#[async_trait]
impl ToolExecutor for WriteChapterOutlineTool {
    fn definition(&self) -> &ToolDefinition {
        &ToolDefinition::new(
            "write_chapter_outline",
            "写入章节梗概",
            "在章节文件中写入剧情梗概",
        )
        .with_input_schema(
            InputSchema::new()
                .property_with_desc("chapter_path", "string", "章节文件路径", true)
                .property_with_desc("outline", "string", "剧情梗概", true)
                .property_with_desc("position", "string", "插入位置 (append/prepend/replace)", false)
        )
    }
    
    async fn execute(&self, input: Value, context: ToolContext) -> ToolResult {
        let path = input["chapter_path"].as_str().unwrap();
        let outline = input["outline"].as_str().unwrap();
        let position = input["position"].as_str().unwrap_or("append");
        
        // 读取现有内容
        let mut content = std::fs::read_to_string(path)
            .unwrap_or_default();
        
        // 插入梗概
        match position {
            "prepend" => {
                content = format!("{}\n\n{}", outline, content);
            }
            "replace" => {
                content = outline.to_string();
            }
            _ => { // append
                content = format!("{}\n\n## 剧情梗概\n\n{}\n", content, outline);
            }
        }
        
        std::fs::write(path, &content)
            .map_err(|e| ToolResult::error(format!("写入失败: {}", e)))?;
        
        ToolResult::success(json!({
            "written": true,
            "path": path,
        }))
    }
}
```

### 2.7 用户批注检测

**目标**: 监听用户在文件中添加的批注

**新增模块**: `packages/sdk/core/src/file_watcher/user_annotation.rs`

```rust
use notify::{Watcher, RecommendedWatcher, RecursiveMode, Event};
use std::sync::Arc;

/// 用户批注监听器
pub struct UserAnnotationWatcher {
    book_id: String,
    mention_handler: Arc<AgentMentionHandler>,
}

impl UserAnnotationWatcher {
    pub fn new(book_id: String, mention_handler: Arc<AgentMentionHandler>) -> Self {
        Self { book_id, mention_handler }
    }
    
    pub async fn start(&self) -> Result<(), WatchError> {
        let book_path = format!("/opennovel/{}", self.book_id);
        
        let mut watcher = RecommendedWatcher::new(
            move |res: Result<Event, notify::Error>| {
                if let Ok(event) = res {
                    if let Some(path) = event.paths.first() {
                        // 检测文件变更
                        if let Err(e) = self.handle_file_change(path) {
                            eprintln!("处理文件变更失败: {}", e);
                        }
                    }
                }
            },
            notify::Config::default(),
        )?;
        
        watcher.watch(book_path.as_ref(), RecursiveMode::Recursive)?;
        
        Ok(())
    }
    
    fn handle_file_change(&self, path: &Path) -> Result<(), WatchError> {
        // 读取文件内容
        let content = std::fs::read_to_string(path)?;
        
        // 检测用户批注格式
        let user_annotation_pattern = r"<!-- 用户批注: (.+?) -->";
        let re = regex::Regex::new(user_annotation_pattern)?;
        
        for cap in re.captures_iter(&content) {
            let user_comment = cap.get(1).unwrap().as_str();
            
            // 触发天道处理用户批注
            self.mention_handler.handle(AgentMention {
                from_agent: "user".to_string(),
                to_agent: "tiandao".to_string(),
                mention_type: MentionType::RequestAction,
                message_id: path.to_string_lossy().to_string(),
                payload: Some(json!({
                    "type": "user_annotation",
                    "content": user_comment,
                })),
            });
        }
        
        Ok(())
    }
}
```

### 2.8 上下文压缩阈值

**目标**: 在 90% 上下文时触发自动压缩

**修改模块**: `packages/sdk/core/src/session/context_manager.rs`

```rust
/// 上下文管理器
pub struct ContextManager {
    /// 最大上下文长度 (GLM-5: 200K)
    max_tokens: usize,
    /// 压缩触发阈值 (90%)
    compression_threshold: f32,
}

impl ContextManager {
    pub fn new() -> Self {
        Self {
            max_tokens: 200_000,
            compression_threshold: 0.9,
        }
    }
    
    /// 检查是否需要压缩
    pub fn should_compress(&self, current_tokens: usize) -> bool {
        let threshold = (self.max_tokens as f32 * self.compression_threshold) as usize;
        current_tokens >= threshold
    }
    
    /// 估算当前上下文 token 数
    pub fn estimate_tokens(&self, messages: &[ChatMessage]) -> usize {
        let mut total = 0;
        for msg in messages {
            // 中文约 1.5 字符/token
            let char_count = msg.content.chars().count();
            total += (char_count as f32 / 1.5) as usize;
        }
        total
    }
    
    /// 执行压缩 (分层保留策略)
    pub fn compress(&self, messages: &mut Vec<ChatMessage>) {
        let mut preserved = Vec::new();
        let mut to_summarize = Vec::new();
        
        for msg in messages {
            // 决策类消息始终保留
            if self.is_decision_message(msg) {
                preserved.push(msg.clone());
            } else {
                to_summarize.push(msg.clone());
            }
        }
        
        // 生成摘要
        let summary = self.generate_summary(&to_summarize);
        
        // 重建消息列表
        *messages = preserved;
        messages.insert(0, ChatMessage::system(&summary));
    }
    
    fn is_decision_message(&self, msg: &ChatMessage) -> bool {
        // 检测用户选择了某个走向等决策类消息
        msg.content.contains("选择走向") || 
        msg.content.contains("确认") ||
        msg.content.contains("同意")
    }
}
```
